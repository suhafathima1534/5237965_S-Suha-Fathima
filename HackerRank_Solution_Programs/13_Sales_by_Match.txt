// SALES BY MATCH - SOLUTION PROGRAM

#include <stdio.h>

int sockMerchant(int n, int ar[]) {
    int count[101] = {0}; // assuming colors are between 1 and 100
    int pairs = 0;

    for (int i = 0; i < n; i++) {
        count[ar[i]]++;
    }

    for (int i = 0; i < 101; i++) {
        pairs += count[i] / 2; // each 2 socks make a pair
    }

    return pairs;
}

int main() {
    int n;
    scanf("%d", &n);
    int ar[n];

    for (int i = 0; i < n; i++) {
        scanf("%d", &ar[i]);
    }

    int result = sockMerchant(n, ar);
    printf("%d\n", result);

    return 0;
}
