#include <math.h>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <assert.h>
#include <limits.h>
#include <stdbool.h>

int main() {
    char s[10001], t[10001], result[10001];
    scanf("%s", s);
    scanf("%s", t);

    int len = strlen(s);
    for (int i = 0; i < len; i++) {
        // XOR logic: if characters are the same, put '0', else '1'
        result[i] = (s[i] == t[i]) ? '0' : '1';
    }
    result[len] = '\0';

    printf("%s\n", result);

    return 0;
}
