// SHERLOCK AND THE VALID STRING - SOLUTION PROGRAM

#include <stdio.h>
#include <string.h>

char* isValid(char *s) {
    int freq[26] = {0};
    int i, n = strlen(s);

    // count frequencies
    for(i = 0; i < n; i++)
        freq[s[i]-'a']++;

    // find non-zero frequencies
    int first = 0, second = 0, f1 = 0, f2 = 0;
    for(i = 0; i < 26; i++) {
        if(freq[i] > 0) {
            if(first == 0) { first = freq[i]; f1++; }
            else if(freq[i] == first) f1++;
            else if(second == 0) { second = freq[i]; f2++; }
            else if(freq[i] == second) f2++;
            else return "NO";
        }
    }

    // check conditions
    if(second == 0) return "YES";  
    if((f2 == 1 && (second - first == 1)) ||   // one char has freq +1
       (f1 == 1 && (first - second == 1)) ||   // one char has freq +1
       (f1 == 1 && first == 1) ||              // one char has freq 1
       (f2 == 1 && second == 1))               // one char has freq 1
        return "YES";

    return "NO";
}

int main() {
    char s[1000];
    scanf("%s", s);
    printf("%s\n", isValid(s));
    return 0;
}
