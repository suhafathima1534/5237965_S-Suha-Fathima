// GRID CHALLENGE - SOLUTION PROGRAM

#include <stdio.h>
#include <string.h>

int main() {
    int t; scanf("%d", &t);
    while (t--) {
        int n; scanf("%d", &n);
        char grid[105][105];

        // read and sort each row
        for (int i = 0; i < n; i++) {
            scanf("%s", grid[i]);
            for (int j = 0; j < n-1; j++)       // bubble sort each row
                for (int k = j+1; k < n; k++)
                    if (grid[i][j] > grid[i][k]) {
                        char tmp = grid[i][j];
                        grid[i][j] = grid[i][k];
                        grid[i][k] = tmp;
                    }
        }

        // check columns
        int ok = 1;
        for (int c = 0; c < n; c++)
            for (int r = 1; r < n; r++)
                if (grid[r][c] < grid[r-1][c]) ok = 0;

        printf(ok ? "YES\n" : "NO\n");
    }
    return 0;
}
