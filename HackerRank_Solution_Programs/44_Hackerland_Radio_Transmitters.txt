// Hackerland Radio Transmitters - Solution Program

#include <stdio.h>
#include <stdlib.h>

int cmp(const void *a, const void *b) {
    return (*(int*)a - *(int*)b);
}

int hackerlandRadioTransmitters(int x[], int n, int k) {
    qsort(x, n, sizeof(int), cmp);  // sort house positions
    int i = 0, count = 0;

    while(i < n) {
        count++;  
        int loc = x[i] + k;         // go as far right as possible
        while(i < n && x[i] <= loc) i++;

        int tower = x[i-1] + k;     // place transmitter
        while(i < n && x[i] <= tower) i++;  // skip covered houses
    }
    return count;
}

int main() {
    int n, k;
    scanf("%d %d", &n, &k);
    int x[n];
    for(int i = 0; i < n; i++) scanf("%d", &x[i]);
    printf("%d\n", hackerlandRadioTransmitters(x, n, k));
    return 0;
}
