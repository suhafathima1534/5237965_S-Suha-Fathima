// LONELY INTEGER - SOLUTION PROGRAM

#include <stdio.h>

int lonelyinteger(int a[], int n) {
    int unique = 0;
    for(int i = 0; i < n; i++) {
        unique ^= a[i]; // XOR all numbers
    }
    return unique;
}

int main() {
    int n;
    scanf("%d", &n);
    int a[n];
    for(int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }
    int result = lonelyinteger(a, n);
    printf("%d\n", result);
    return 0;
}
