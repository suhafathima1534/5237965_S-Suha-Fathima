// SUBARRAY DIVISION 1

#include <stdio.h>

// Function to find the number of ways to divide the chocolate
int birthday(int s[], int n, int d, int m) {
    int count = 0;
    for(int i = 0; i <= n - m; i++) {
        int sum = 0;
        // Sum up m consecutive squares
        for(int j = 0; j < m; j++) {
            sum += s[i + j];
        }
        if(sum == d) {
            count++; // Found a valid way
        }
    }
    return count;
}

int main() {
    int n;
    scanf("%d", &n); // Number of squares

    int s[100]; // Array to hold squares (max size given in constraints)
    for(int i = 0; i < n; i++) {
        scanf("%d", &s[i]); // Read integers on the squares
    }

    int d, m;
    scanf("%d %d", &d, &m); // Ron's birth day and birth month

    int result = birthday(s, n, d, m);
    printf("%d\n", result);

    return 0;
}
