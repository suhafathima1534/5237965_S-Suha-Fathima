#include <stdio.h>

void minimumBribes(int q[], int n) {
    int bribes = 0;
    int chaotic = 0;

    for (int i = n - 1; i >= 0; i--) {
        // If person has moved more than 2 positions ahead
        if (q[i] - (i + 1) > 2) {
            printf("Too chaotic\n");
            chaotic = 1;
            break;
        }
        // Count number of people who bribed this person
        // Start from one position ahead of their initial position or 0, whichever is larger
        for (int j = (q[i] - 2 > 0 ? q[i] - 2 : 0); j < i; j++) {
            if (q[j] > q[i]) {
                bribes++;
            }
        }
    }
    if (!chaotic) {
        printf("%d\n", bribes);
    }
}

int main() {
    int t;
    scanf("%d", &t);

    while (t--) {
        int n;
        scanf("%d", &n);
        int q[n];
        for (int i = 0; i < n; i++) {
            scanf("%d", &q[i]);
        }
        minimumBribes(q, n);
    }
    return 0;
}
