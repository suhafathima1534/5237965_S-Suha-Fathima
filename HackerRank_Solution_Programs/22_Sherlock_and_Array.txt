// SHERLOCK AND ARRAY - SOLUTION PROGRAM

#include <stdio.h>

int main() {
    int T;
    scanf("%d", &T);
    while (T--) {
        int n;
        scanf("%d", &n);
        int arr[n];
        long long sum = 0, left = 0;
        for (int i = 0; i < n; i++) {
            scanf("%d", &arr[i]);
            sum += arr[i];
        }
        int found = 0;
        for (int i = 0; i < n; i++) {
            if (left == sum - left - arr[i]) {
                found = 1;
                break;
            }
            left += arr[i];
        }
        printf("%s\n", found ? "YES" : "NO");
    }
    return 0;
}
